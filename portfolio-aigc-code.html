 # Python 小工具示例 (AIGC辅助生成)

 import os # 导入os模块，用于文件操作
 # import shutil # 如果需要复制或移动文件，导入shutil

 def find_large_files(directory, min_size_mb):
     """
     查找指定目录下大于等于 min_size_mb 的文件。

     Args:
         directory (str): 要查找的目录路径。
         min_size_mb (int): 文件大小的最小阈值（MB）。

     Returns:
         list: 包含符合条件的文件路径和大小的列表。
     """
     large_files = [] # 初始化空列表存放结果
     min_size_bytes = min_size_mb * 1024 * 1024 # 将MB转换为Bytes

     # 遍历目录及子目录
     for root, _, files in os.walk(directory):
         for file in files:
             file_path = os.path.join(root, file) # 获取完整文件路径
             try:
                 # 获取文件大小
                 file_size_bytes = os.path.getsize(file_path)
                 if file_size_bytes >= min_size_bytes:
                     file_size_mb_actual = file_size_bytes / (1024 * 1024) # 实际大小MB
                     large_files.append((file_path, f"{file_size_mb_actual:.2f} MB")) # 添加到结果列表
             except Exception as e:
                 # 处理访问文件可能出现的异常
                 print(f"Error accessing file {file_path}: {e}")

     return large_files # 返回结果列表

 # 示例用法
 if __name__ == "__main__":
     # 请替换 '/path/to/your/directory' 为实际目录
     target_directory = './downloads' # 假设查找当前目录下的downloads文件夹
     threshold_mb = 100 # 查找大于等于100MB的文件

     print(f"正在查找目录 '{target_directory}' 下大于等于 {threshold_mb}MB 的文件...")
     found_files = find_large_files(target_directory, threshold_mb)

     if found_files:
         print("\n找到以下大文件:")
         for file, size in found_files:
             print(f"- {file} ({size})")
     else:
         print("\n未找到符合条件的大文件。")
